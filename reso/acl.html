Access Control List

Access List (ACL) adalah daftar aturan yang digunakan pada perangkat jaringan untuk mengatur lalu lintas data yang melewati interface. ACL dapat memeriksa paket berdasarkan kriteria tertentu, seperti alamat IP, protokol, atau nomor port. Berdasarkan pemeriksaan tersebut, paket akan diizinkan (permit) atau ditolak (deny) sesuai kebijakan. ACL digunakan untuk meningkatkan keamanan, membatasi akses, dan mengoptimalkan lalu lintas jaringan.

Jenis-jenis ACL

  • Standard ACL – Menyaring berdasarkan alamat IP sumber saja.
  • Extended ACL – Menyaring berdasarkan IP sumber, IP tujuan, protokol, dan port.
  • Named ACL – Menggunakan nama untuk pengelolaan aturan yang lebih fleksibel.

Manfaat ACL

  • Meningkatkan keamanan jaringan dengan membatasi akses tidak sah.
  • Mengurangi lalu lintas yang tidak diperlukan.

    • Konfigurasi supaya PC 2 tidak bisa melakukan ping ke :
    • 192.168.100.252 & 253 dengan ACL Standard
    • 192.168.200.252 & 253 dengan ACL Extended

# Membuat VLAN interface di atas bridge BRIDGE_VLAN
/interface vlan
add interface=BRIDGE_VLAN name=vlan100 vlan-id=100   # VLAN 100 untuk segmen jaringan 1
add interface=BRIDGE_VLAN name=vlan200 vlan-id=200   # VLAN 200 untuk segmen jaringan 2

# Menambahkan port fisik ke bridge
/interface bridge port
add bridge=BRIDGE_VLAN interface=ether24             # ether24 sebagai trunk (tagged VLAN 100 & 200)
add bridge=BRIDGE_VLAN interface=ether1 pvid=100     # ether1 akses VLAN 100
add bridge=BRIDGE_VLAN interface=ether2 pvid=100     # ether2 akses VLAN 100
add bridge=BRIDGE_VLAN interface=ether3 pvid=100     # ether3 akses VLAN 100
add bridge=BRIDGE_VLAN interface=ether4 pvid=200     # ether4 akses VLAN 200
add bridge=BRIDGE_VLAN interface=ether5 pvid=200     # ether5 akses VLAN 200
add bridge=BRIDGE_VLAN interface=ether6 pvid=200     # ether6 akses VLAN 200

# Mapping VLAN ID dengan port yang tagged dan untagged di bridge
/interface bridge vlan
add bridge=BRIDGE_VLAN tagged=ether24 untagged=ether1,ether2,ether3 vlan-ids=100   # VLAN 100: trunk di ether24, untagged di ether1-3
add bridge=BRIDGE_VLAN tagged=ether24 untagged=ether4,ether5,ether6 vlan-ids=200   # VLAN 200: trunk di ether24, untagged di ether4-6

# Mengatur nama perangkat (hostname)
/system identity
set name=CRS.DITRIBUTE

! Mengatur nama router
hostname R1

! IP yang tidak boleh diberikan oleh DHCP (reservasi atau exclusion)
ip dhcp excluded-address 192.168.100.2 192.168.100.253   ! VLAN 100: IP 2-253 tidak dibagikan DHCP
ip dhcp excluded-address 192.168.200.2 192.168.200.253   ! VLAN 200: IP 2-253 tidak dibagikan DHCP

! DHCP pool untuk link antar router (LAN point-to-point)
ip dhcp pool DHCP_LAN
   network 192.168.1.0 255.255.255.252   ! Jaringan /30 untuk antar router
   default-router 192.168.1.1            ! Gateway sisi R1

! DHCP pool untuk VLAN 100
ip dhcp pool DHCP_VLAN100
   network 192.168.100.0 255.255.255.0   ! Jaringan VLAN 100 /24
   default-router 192.168.100.1          ! Gateway VLAN 100

! DHCP pool untuk VLAN 200
ip dhcp pool DHCP_VLAN200
   network 192.168.200.0 255.255.255.0   ! Jaringan VLAN 200 /24
   default-router 192.168.200.1          ! Gateway VLAN 200

! Interface fisik Fa0/0 digunakan sebagai trunk VLAN
interface FastEthernet0/0
 no ip address                           ! Tidak pakai IP langsung, hanya trunk
 duplex half                             ! Mode half-duplex

! Subinterface untuk VLAN 100
interface FastEthernet0/0.100
 encapsulation dot1Q 100                 ! VLAN ID 100
 ip address 192.168.100.1 255.255.255.0  ! IP gateway VLAN 100

! Subinterface untuk VLAN 200
interface FastEthernet0/0.200
 encapsulation dot1Q 200                 ! VLAN ID 200
 ip address 192.168.200.1 255.255.255.0  ! IP gateway VLAN 200
 ip access-group 100 in                  ! Filter ACL 100 untuk traffic masuk VLAN 200

! Interface ke jaringan point-to-point (ISP atau router lain)
interface FastEthernet1/0
 ip address 192.168.1.1 255.255.255.252  ! IP sisi R1
 ip access-group 1 out                   ! ACL 1 untuk traffic keluar
 duplex half                             ! Mode half-duplex

! ACL 1: memblokir IP tertentu dari keluar ke Fa1/0
access-list 1 deny   192.168.100.253     ! Blok IP 192.168.100.253
access-list 1 permit any                 ! Izinkan semua selain itu

! ACL 100: memblokir ICMP dari host VLAN 200 ke host di jaringan /30
access-list 100 deny   icmp host 192.168.200.252 host 192.168.1.2
access-list 100 deny   icmp host 192.168.200.253 host 192.168.1.2
access-list 100 permit ip any any        ! Izinkan semua trafik lain
