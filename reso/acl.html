Access Control List

Access List (ACL) adalah daftar aturan yang digunakan pada perangkat jaringan untuk mengatur lalu lintas data yang melewati interface. ACL dapat memeriksa paket berdasarkan kriteria tertentu, seperti alamat IP, protokol, atau nomor port. Berdasarkan pemeriksaan tersebut, paket akan diizinkan (permit) atau ditolak (deny) sesuai kebijakan. ACL digunakan untuk meningkatkan keamanan, membatasi akses, dan mengoptimalkan lalu lintas jaringan.

Jenis-jenis ACL

  • Standard ACL – Menyaring berdasarkan alamat IP sumber saja.
  • Extended ACL – Menyaring berdasarkan IP sumber, IP tujuan, protokol, dan port.
  • Named ACL – Menggunakan nama untuk pengelolaan aturan yang lebih fleksibel.

Manfaat ACL

  • Meningkatkan keamanan jaringan dengan membatasi akses tidak sah.
  • Mengurangi lalu lintas yang tidak diperlukan.

    • CONTOH KASUS

    • Tujuan: Membatasi akses host tertentu dari tiap VLAN menggunakan ACL (Standard dan Extended).
    • ACL Standard (ACL 1):
      • Memblokir host 192.168.100.252 dari VLAN 100 agar tidak bisa keluar melalui interface Fa1/0.
      • Diterapkan pada arah outbound di interface FastEthernet1/0.
    • ACL Extended (ACL 100):
      • Memblokir ICMP dari host 192.168.200.252 (VLAN 200) ke alamat 192.168.1.2 di jaringan /30.
      • Diterapkan pada arah inbound di subinterface FastEthernet0/0.200.
    • ACL Extended Named (ACL_NAMED):
      • Memblokir ICMP dari host 192.168.210.252 (VLAN 210) ke alamat 192.168.1.2 di jaringan /30.
      • Diterapkan pada arah inbound di subinterface FastEthernet0/0.210.
    • Catatan:
      • ACL Standard memfilter hanya berdasarkan alamat sumber.
      • ACL Extended bisa memfilter berdasarkan sumber, tujuan, dan protokol (contohnya ICMP).
      • Setiap VLAN memiliki gateway pada R1 dan dikontrol oleh ACL masing-masing.

# Membuat bridge dengan VLAN filtering aktif
/interface bridge
add name=BRIDGE_VLAN vlan-filtering=yes

# Membuat VLAN interface di atas bridge BRIDGE_VLAN
/interface vlan
add interface=BRIDGE_VLAN name=vlan100 vlan-id=100   # VLAN 100 untuk segmen jaringan 1
add interface=BRIDGE_VLAN name=vlan200 vlan-id=200   # VLAN 200 untuk segmen jaringan 2
add interface=BRIDGE_VLAN name=vlan210 vlan-id=210   # VLAN 210 untuk segmen jaringan 3

# Menambahkan port fisik ke dalam bridge
/interface bridge port
add bridge=BRIDGE_VLAN interface=ether24             # ether24 sebagai trunk (tagged VLAN 100,200,210)
add bridge=BRIDGE_VLAN interface=ether1 pvid=100     # ether1 akses VLAN 100 (untagged)
add bridge=BRIDGE_VLAN interface=ether2 pvid=100     # ether2 akses VLAN 100 (untagged)
add bridge=BRIDGE_VLAN interface=ether3 pvid=200     # ether3 akses VLAN 200 (untagged)
add bridge=BRIDGE_VLAN interface=ether4 pvid=200     # ether4 akses VLAN 200 (untagged)
add bridge=BRIDGE_VLAN interface=ether5 pvid=210     # ether5 akses VLAN 210 (untagged)
add bridge=BRIDGE_VLAN interface=ether6 pvid=210     # ether6 akses VLAN 210 (untagged)

# Mapping VLAN ID dengan port tagged (trunk) dan untagged (akses)
/interface bridge vlan
add bridge=BRIDGE_VLAN tagged=ether24 untagged=ether1,ether2 vlan-ids=100   # VLAN 100: trunk ether24, akses ether1-2
add bridge=BRIDGE_VLAN tagged=ether24 untagged=ether3,ether4 vlan-ids=200   # VLAN 200: trunk ether24, akses ether3-4
add bridge=BRIDGE_VLAN tagged=ether24 untagged=ether5,ether6 vlan-ids=210   # VLAN 210: trunk ether24, akses ether5-6

# Mengatur nama perangkat (hostname)
/system identity
set name=CRS.DITRIBUTE

! Interface fisik Fa0/0 sebagai trunk (tidak pakai IP langsung)
interface FastEthernet0/0
 no ip address
 duplex half

! Subinterface untuk VLAN 100
interface FastEthernet0/0.100
 encapsulation dot1Q 100                  ! VLAN ID 100
 ip address 192.168.100.1 255.255.255.0   ! Gateway VLAN 100

! Subinterface untuk VLAN 200
interface FastEthernet0/0.200
 encapsulation dot1Q 200                  ! VLAN ID 200
 ip address 192.168.200.1 255.255.255.0   ! Gateway VLAN 200
 ip access-group 100 in                   ! Terapkan ACL 100 ke arah masuk VLAN 200

! Subinterface untuk VLAN 210
interface FastEthernet0/0.210
 encapsulation dot1Q 210                  ! VLAN ID 210
 ip address 192.168.210.1 255.255.255.0   ! Gateway VLAN 210
 ip access-group ACL_NAMED in             ! Terapkan ACL_NAMED ke arah masuk VLAN 210

! Interface point-to-point ke router/ISP lain
interface FastEthernet1/0
 ip address 192.168.1.1 255.255.255.252   ! IP sisi R1 pada link /30
 ip access-group 1 out                    ! Terapkan ACL 1 untuk traffic keluar
 duplex half

! ACL NAMED: memblokir ICMP dari host VLAN 210 ke host di jaringan /30
ip access-list extended ACL_NAMED
 deny   icmp host 192.168.210.252 host 192.168.1.2
 permit ip any any

! ACL 1: memblokir IP tertentu dari VLAN 100 keluar ke Fa1/0
access-list 1 deny   192.168.100.252
access-list 1 permit any

! ACL 100: memblokir ICMP dari host VLAN 200 ke host di jaringan /30
access-list 100 deny   icmp host 192.168.200.252 host 192.168.1.2
access-list 100 permit ip any any
